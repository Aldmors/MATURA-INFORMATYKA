def czy_k_podobne(n, A, B, k):
    for i in range(k):

        if A[i] != B[n - k + i]:
            return False

    for i in range(n - k):
        if B[i] != A[k + i]:
            return False
    return True


def czy_podobne(n, A, B):
    for k in range(n):
        if czy_k_podobne(n, A, B, k):
            return True
        return False


print(czy_k_podobne(n, A, B, k))

def is_prime(x):
    if x > 1:
        for i in range(2, x):
            if x % i == 0:
                return False
        return True
    return False

    for line in file:
        line, slowo = line.strip().split()
        res = slowo[0]
        longest = slowo[0]
        for i in range(1, len(slowo)):
            if slowo[i - 1] == slowo[i]:
                res += slowo[i - 1]
            else:
                res = slowo[i]
            if len(res) > len(longest):
                longest = res
        print(longest, len(longest))
