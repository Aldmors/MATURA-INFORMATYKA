```py
def prime_factors(n):
    i = 2
    factors = set()
    while i * i <= n:
        if n % i:
            i += 1
        else:
            n //= i
            factors.add(i)
    if n > 1:
        factors.add(n)
    return factors


count = 0

with open("dane/59/liczby.txt") as file:
    for line in file:
        line = int(line.strip())
        liczby = prime_factors(line)

        if len(liczby) == 3 and all(i % 2 != 0 for i in liczby):
            count += 1

print(count)
```
