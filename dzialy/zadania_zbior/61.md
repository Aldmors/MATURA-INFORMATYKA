liczby = []

with open("dane/61/ciagi.txt") as file:
    for line in file:
        liczby.append(line.strip())

ciagi = []

for i in range(1,len(liczby),2):
    ciagi.append(liczby[i])

liczby.clear()

count = 0
maxi = -1
for i in ciagi:
    i = list(map(int,i.split()))
    r = abs(i[1] - i[0])
    if all(i[j] == i[j-1] + r for j in range(1,len(i))):
        count += 1
        if r > maxi:
            maxi = r

print(count)
print(maxi)
